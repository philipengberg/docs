---
title: 'Get electricity prices'
api: 'GET /thirdParty/prices/{region}?addressId={addressId}'
---
<Accordion title="Parameters">
<ParamField path="region" type="string" required="true">
  See [Regions](/api-reference/shared#regions) for possible values
</ParamField>
<ParamField query="addressId" type="string">
  The ID of a connected address. If available, this will include all charges for the address in the electricity price.
</ParamField>
</Accordion>

<Accordion title="Response">
<ResponseField name="username" type="string" required>
  The username of the user
</ResponseField>
</Accordion>
